                function exponentiation(x, y) {
                    let result = 1;
                    for (let i = 1; i <= y; i++) {
                        result = result * x;
                    }
                    let formatNumber = new Intl.NumberFormat('ru');
                    result = formatNumber.format(result);
                    return console.log(`${x} ^ ${y} = ${result}`);
                }

                function multiplication(x, y) {
                    let result = 0;
                    for (let i = 1; i <= y; i++) {
                        result = result + x;
                    }
                    let formatNumber = new Intl.NumberFormat('ru');
                    x = formatNumber.format(x);
                    y = formatNumber.format(y);
                    result = formatNumber.format(result);
                    return console.log(`${x} * ${y} = ${result}`);
                }
